using Xunit;
using Moq;

public class RefreshManagerTests
{
    private RefreshManager CreateSut()
    {
        // Clear localStorage before each test
        localStorage.clear();
        return new RefreshManager();
    }

    [Fact]
    public void InitialState_LoadsDefaultInterval()
    {
        var sut = CreateSut();
        
        Assert.Equal(10000, sut.interval);
        Assert.False(sut.isPaused);
        Assert.Equal(10, sut.timeRemaining);
    }

    [Fact]
    public void SetInterval_UpdatesLocalStorageAndRestartsCountdown()
    {
        var mockFetchCallback = new Mock<Func<Task>>();
        var sut = CreateSut();
        
        sut.start(mockFetchCallback.Object);
        sut.setInterval(30000);

        Assert.Equal(30000, sut.interval);
        Assert.Equal(30, sut.timeRemaining);
        mockFetchCallback.Verify(f => f(), Times.Once);
    }

    [Fact]
    public void Pause_StopsCountdown()
    {
        var sut = CreateSut();
        sut.pause();

        Assert.True(sut.isPaused);
        Assert.Equal("⏸️", sut.updateCountdownDisplay());
    }

    [Fact]
    public void Resume_RestartsCountdown()
    {
        var sut = CreateSut();
        sut.pause();
        sut.resume();

        Assert.False(sut.isPaused);
        Assert.Equal(10, sut.timeRemaining);
    }
}`,

using Xunit;
using Moq;

public class NotificationManagerTests
{
    private NotificationManager CreateSut()
    {
        return new NotificationManager();
    }

    [Fact]
    public void DetectChanges_TriggersNotificationsForStateTransitions()
    {
        var sut = CreateSut();
        var mockToastManager = new Mock<ToastManager>();
        var mockSoundManager = new Mock<SoundManager>();

        var initialData = new {
            stories = new[] {
                new { 
                    workItemId = 123, 
                    agents = new Dictionary<string, string> {
                        { "Planning", "pending" },
                        { "Coding", "pending" }
                    }
                }
            }
        };

        var updatedData = new {
            stories = new[] {
                new { 
                    workItemId = 123, 
                    agents = new Dictionary<string, string> {
                        { "Planning", "completed" },
                        { "Coding", "active" }
                    }
                }
            }
        };

        sut.detectChanges(initialData);
        sut.detectChanges(updatedData);

        mockToastManager.Verify(
            t => t.show(
                It.Is<string>(s => s == "completed"),
                It.Is<string>(s => s.Contains("Planning")),
                It.IsAny<string>()
            ), 
            Times.Once
        );
    }

    [Fact]
    public void SoundSettings_RespectUserPreferences()
    {
        var mockSettingsManager = new Mock<SettingsManager>();
        mockSettingsManager
            .Setup(s => s.get("soundEnabled"))
            .Returns(false);

        var mockSoundManager = new Mock<SoundManager>();
        var sut = new NotificationManager(
            mockSettingsManager.Object, 
            mockSoundManager.Object
        );

        sut.notify("Planning", 123, "completed");

        mockSoundManager.Verify(
            s => s.playSound(It.IsAny<string>()),
            Times.Never
        );
    }
}`,
